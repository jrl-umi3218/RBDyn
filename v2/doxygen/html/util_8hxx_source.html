7template<typename T>
8std::ostream & operator<<(std::ostream & o, const std::vector<T> & vect)
9{
10 typedef typename std::vector<T>::const_iterator citer_t;
11
12 if(vect.empty()) return o << "Empty vector";
13
14 citer_t it = vect.begin();
15 o << "[ " << *it;
16 ++it;
17
18 for(; it != vect.end(); ++it) o << ", " << *it;
19 o << "]";
20 return o;
21}

44template<typename T>
45std::ostream & operator<<(std::ostream & o, const Eigen::MatrixBase<T> & matrix)
46{
47 Eigen::IOFormat ioformat(Eigen::StreamPrecision, Eigen::DontAlignCols, ",", ", ", "(", ")", "(", ")");
48 ioformat.rowSpacer = "";
49 o << "[";
50
51 // Matrix
52 if(matrix.cols() == 1 || matrix.rows() == 1)
53 {
54 // Vector
55 ioformat = Eigen::IOFormat(Eigen::StreamPrecision, Eigen::DontAlignCols, ",", ",", "", "", "(", ")");
56 ioformat.rowSpacer = "";
57 o << matrix.size();
58 }
59 else
60 o << matrix.rows() << "," << matrix.cols();
61
62 o << "]" << matrix.format(ioformat);
63 return o;

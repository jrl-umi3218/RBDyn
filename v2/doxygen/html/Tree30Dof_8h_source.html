102std::tuple<rbd::MultiBody, rbd::MultiBodyConfig, rbd::MultiBodyGraph> makeTree30Dof(bool isFixed = true)
103{
104 using namespace Eigen;
105 using namespace sva;
106 using namespace rbd;
107
108 MultiBodyGraph mbg;
109
110 double mass = 1.;
111 Matrix3d I = Matrix3d::Identity();
112 Vector3d h = Vector3d::Zero();
113
114 RBInertiad rbi(mass, h, I);
115
116 mbg.addBody({rbi, "BODY0"});
117 mbg.addBody({rbi, "BODY1"});
118 mbg.addBody({rbi, "TORSO"});
119 mbg.addBody({rbi, "HEAD0"});
120 mbg.addBody({rbi, "HEAD1"});
121
122 mbg.addJoint({Joint::RevY, true, "BODY0_BODY1"});
123 mbg.addJoint({Joint::RevX, true, "BODY1_TORSO"});
124 mbg.addJoint({Joint::RevY, true, "TORSO_HEAD0"});
125 mbg.addJoint({Joint::RevX, true, "HEAD0_HEAD1"});
126
127 PTransformd to(Vector3d(0., 0.1, 0.));
128 PTransformd from(PTransformd::Identity());
129
130 mbg.linkBodies("BODY0", to, "BODY1", from, "BODY0_BODY1");
131 mbg.linkBodies("BODY1", to, "TORSO", from, "BODY1_TORSO");
132 mbg.linkBodies("TORSO", to, "HEAD0", from, "TORSO_HEAD0");
133 mbg.linkBodies("HEAD1", to, "HEAD1", from, "HEAD0_HEAD1");
134
135 // left arm
136 createArm(mbg, Vector3d(-0.1, 0.05, 0.), "TORSO", "L");
137 // right arm
138 createArm(mbg, Vector3d(0.1, 0.05, 0.), "TORSO", "R");
139 // left leg
140 createLeg(mbg, Vector3d(-0.1, -0.05, 0.), "BODY0", "L");
141 // right leg
142 createLeg(mbg, Vector3d(0.1, -0.05, 0.), "BODY0", "R");
143
144 MultiBody mb = mbg.makeMultiBody("BODY0", isFixed);
145
146 MultiBodyConfig mbc(mb);
147 mbc.zero(mb);
148
149 return std::make_tuple(mb, mbc, mbg);
150}
